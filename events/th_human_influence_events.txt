# 争夺结果trigger
country_event = {
	id = huminf.4

	hidden = yes

	fire_only_once = no
	is_triggered_only = yes

	option = {
		calc_human_influence = yes
		for_each_scope_loop = {
			array = HUM.possibleInfluContries
			if = {
				limit = {
					OR = {
						has_idea = hum_ren_lei_cheng_fa_zhan_zheng_de_sheng_li_zhe_idea
						original_tag = HUM
					}
				}
				country_event = {
					id = huminf.3
				}
			}
		}
		if = {
			limit = {
				check_variable = {
					var = HUM.reserveInfluence
					value = 75
					compare = greater_than_or_equals
				}
			}
			news_event = huminf.5
		}
	}
}

	is_triggered_only = yes
