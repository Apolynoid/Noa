country_event = { #一个人类流亡者被消灭
	id = hud.4
	title = hud.4.t
	desc = hud.4.d

	fire_only_once = no
	is_triggered_only = no
	mean_time_to_happen = {
		days = 5
	}

	trigger = {
		tag = HUM
		has_global_flag = main_scenario
		not = {
			has_global_flag = HUV_all_extinct
		}
		or = {
			not = {
				or = {
					has_global_flag = HU6_extinct
					country_exists = HU6
				}
			}
			not = {
				or = {
					has_global_flag = HU7_extinct
					country_exists = HU7
				}
			}
			not = {
				or = {
					has_global_flag = HU8_extinct
					country_exists = HU8
				}
			}
		}
	}

	option = {
		name = hud.4.1
		if = {
			limit = {
				not = {
					or = {
						has_global_flag = HU8_extinct
						country_exists = HU8
					}
				}
			}
			set_global_flag = HU8_extinct
			every_state = {
				remove_core_of = HU8
				remove_claim_by = HU8
			}
		}
		if = {
			limit = {
				not = {
					or = {
						has_global_flag = HU7_extinct
						country_exists = HU7
					}
				}
			}
			set_global_flag = HU7_extinct
			every_state = {
				remove_core_of = HU7
				remove_claim_by = HU7
			}
		}
		if = {
			limit = {
				not = {
					or = {
						has_global_flag = HU6_extinct
						country_exists = HU6
					}
				}
			}
			set_global_flag = HU6_extinct
			every_state = {
				remove_core_of = HU6
				remove_claim_by = HU6
			}
		}
		if = {
			limit = {
				has_global_flag = HU6_extinct
				has_global_flag = HU7_extinct
				has_global_flag = HU8_extinct
			}
			hidden_effect = {
				news_event = { id = hud.5 days = 2}
			}
		}
	}
}
